class User
{
    public $name;
    protected $email;
    private $password;

    public function setEmail($newEmail)
    {
        //Prueft ob $newEmail ueber die noch folgende Methode isValidEmail($email) gueltig ist...
        if ($this->isValidEmail($newEmail)) {
            //...wen ja, setze die Email
            $this->email = $newEmail;
        }
    }

    public function getEmail()
    {
        //Wirft die Email-Adresse zurueck
        return $this->email;
    }

    public function setRandomPassword()
    {
        //Erstellt eine zufaellige Zahl zwischen 1 und 10
        $randomPassword = strval(rand(1, 10));
        //Speichert die Zahl als $password, nachdem sie durch die noch folgende Funktion hashPassword in sha1
        //verschluesselt wurde
        $this->password = $this->hashPassword($randomPassword);

        return $randomPassword;
    }

    public function isCorrectPassword($password)
    {
        //Gibt zurueck ob $password dem Klassen-Password entspricht
        return $this->hashPassword($password) == $this->password;
    }

    protected function isValidEmail($email)
    {
        //Prueft, ab es sich um eine gueltige Email-Adresse handelt, wenn ja, wird sie zurueckgeworfen...
        //...filter_var ueberprueft die uebergebende Variable dabei mit dem Filter FILTER_VALIDATE_EMAIL
        return filter_var($email, FILTER_VALIDATE_EMAIL) !==false;
    }

    private function hashPassword($password)
    {
        //Gibt das $password zurueck, nachdem es mit sha1 verschluesselt wurde
        return sha1($password);
    }
}

$simon = new User();
$simon->name = "Simon Weis";
$simon->setEmail("simon@web.de");

echo "Die Email-Adresse von $simon->name ist: " . $simon->getEmail() . "
";

$randomPW = $simon->setRandomPassword();

echo "Zufaelliges Passwort: $randomPW
