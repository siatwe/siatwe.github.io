Binary search works on sorted arrays. Binary search begins by comparing the middle element of the array with the target value. If the target value matches the middle element, its position in the array is returned. If the target value is less than or greater than the middle element, the search continues in the lower or upper half of the array, respectively, eliminating the other half from consideration.

Class Number

require_once "Terminal.php";

class Number
{

    private $numberRange;
    private $determinedNumber;
    private $numberRangeArray;


    private function determine($numberRange)
    {
        $this->determinedNumber = mt_rand(1, $numberRange);
    }


    public function check($guessedNumber)
    {
        return ($guessedNumber === $this->determinedNumber) ? true : false;
    }


    public function determineNumberRange($numberRange)
    {
        if ($numberRange > 799 || $numberRange < 1) {
            echo "\e[41mInvalid input! Press return and try again\e[0m";
            $dump = readline();
            Terminal::start();
        } else {
            $this->numberRange = $numberRange;
            $this->generateNumberRangeArray();
            $this->determine($numberRange);
        }
    }


    private function generateNumberRangeArray()
    {
        for ($i = 0; $i < $this->numberRange; $i++) {
            $this->numberRangeArray[$i] = $i;
        }
    }


    public function guessANumber($guessedNumber)
    {
        if ($guessedNumber === $this->determinedNumber) {
            $this->numberRangeArray[$guessedNumber] = "\e[45m".$guessedNumber."\e[0m";
        } else {
            if ($guessedNumber < $this->determinedNumber) {
                for (; $guessedNumber >= 0; $guessedNumber--) {
                    $this->numberRangeArray[$guessedNumber] = "\e[91m".$guessedNumber."\e[0m";
                }
            } else {
                $newRange = $guessedNumber;
                for (; $guessedNumber <= $this->numberRange; $guessedNumber++) {
                    $this->numberRangeArray[$guessedNumber] = "\e[91m".$guessedNumber."\e[0m";
                }
                $this->numberRange = $newRange;
            }
        }
        Terminal::drawNumberRange($this->numberRangeArray);
    }


    public function getNumberRange()
    {
        return $this->numberRange;
    }


    public function getNumberRangeArray()
    {
        return $this->numberRangeArray;
    }


    public function getDeterminedNumber()
    {
        return $this->determinedNumber;
    }
}

Class Terminal

require_once "Number.php";

class Terminal
{
    private $firstRun = 0;


    public static function start()
    {
        echo "\033[H\033[2J";
        echo "Number range (0 - 799): ";
        $number = new Number();
        $number->determineNumberRange(readline());
        self::drawNumberRange($number->getNumberRangeArray());
        $guessedNumber = -1;
        while ($guessedNumber != $number->getDeterminedNumber()) {
            echo "\nYour guess: ";
            $guessedNumber = readline();
            $number->guessANumber($guessedNumber);
        }
        echo "\nYes, ".$guessedNumber." was my determined number!\n";
    }


    public static function drawNumberRange($numberRange)
    {
        echo "\033[H\033[2JBinary Search Example\n";
        $break = 0;
        foreach ($numberRange as $number) {
            echo "\e[92m".$number."\e[0m ";
            $break++;
            if ($break === 25) {
                echo "\n";
                $break = 0;
            }
        }
    }
}
