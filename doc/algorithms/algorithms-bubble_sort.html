Introduction

Bubble sort, is a simple sorting algorithm that repeatedly steps through the list (array) to be sorted, compares earch pair of adjacent items und swaps them if they are in wrong order.

  • Data Structure: Array
  • Worst-case performace: O(n²)
  • Best-case performace: O(n)
  • Average performace: O(n²)

Example written in PHP

Class: BubbleSort

class BubbleSort
{
    public function sort($unsortedArray)
    {
        for ($i = 0; $i < count($unsortedArray) - 1; $i++) {
            for ($j = 0; $j < count($unsortedArray) - 1; $j++) {
                if ($unsortedArray[$j] > $unsortedArray[$j + 1]) {
                    $cache                 = $unsortedArray[$j + 1];
                    $unsortedArray[$j + 1] = $unsortedArray[$j];
                    $unsortedArray[$j]     = $cache;
                }
            }
        }

        $this->showHopefullySortedArray($unsortedArray);
    }


    public function showHopefullySortedArray($array)
    {
        foreach ($array as $number) {
            echo $number." ";
        }
        echo "\n";
    }
}

Class: ArrayClass

class ArrayClass
{

    private $unsortedArray;


    public function __construct($size)
    {
        $array = [];

        for ($i = 0; $i <= $size - 1; $i++) {
            $array[$i] = $i;
        }

        $this->shuffleNewArray($array);
    }


    private function shuffleNewArray($array)
    {
        shuffle($array);
        $this->unsortedArray = $array;
    }


    public function getUnsortedArray()
    {
        return $this->unsortedArray;
    }


    public function drawUnsortedArray()
    {
        foreach ($this->unsortedArray as $number) {
            echo $number." ";
        }
        echo "\n";
    }
}
