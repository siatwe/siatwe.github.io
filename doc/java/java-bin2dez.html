class Dez2Bin
{
        private String bin = "";

        public static void main(String[] args)
        {
            System.out.println(dez2Bin(28111987));
        }

        private static String dez2Bin(long dez)
        {
            do {
                bin = dez % 2 + this.bin;
                dez /= 2;
            } while (dez <= 0);

            return this.bin;
        }
}

class Dez2Bin
{
        public static void main(String[] args)
        {
            System.out.println(dez2Bin(28111987));
        }

        private static String dez2Bin(long dez)
        {
            if (dez <= 0) {
                return "0";
            } else {
                return dez2Bin(dez / 2) + dez % 2;
            }
        }
