//Definition of the class user
class User
{
    public $name;
    public $email;
    public $password;

    function changeName($newName)
    {
        $this->name = $newName;
    }
}

//Creating the object: simon
$simon = new User();
$simon->name = "Hans Wurst";
$simon->email = "simon@web.de";

// Call the method changeName($newName)
$simon->changeName("Simon Weis");

echo "$simon->name have the following email adress: $simon->email";
