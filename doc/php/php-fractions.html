class Fraction
{
    public $numerator;
    public $denominator;


    public function __construct($numerator, $denominator)
    {
        $this->numerator   = $numerator;
        $this->denominator = $denominator;
    }


    public function printFraction()
    {
        echo $this->numerator. "/" . $this->denominator . " ";
    }
}

public function greatestCommonDivisorOn($a, $b)
{
    $maxDivisor = ($a > $b) ? $a : $b;
    $divisor    = 1;
    for ($i = 2; $i <= $maxDivisor; $i++) {
        if ((($a % $i) + ($b % $i)) === 0) {
            $divisor = $i;
        }
    }

    return $divisor;
}

public function lowestCommonDenominatorOn(Fraction $a, Fraction $b)
{
    if (!($a->denominator === $b->denominator)) {

        $lowestCommonDenominator = $a->denominator * $b->denominator;

        $a->numerator = $a->numerator * $b->denominator;
        $b->numerator = $b->numerator * $a->denominator;

        $a->denominator = $lowestCommonDenominator;
        $b->denominator = $lowestCommonDenominator;
    }

    $fractions['a'] = new Fraction($a->numerator, $a->denominator);
    $fractions['b'] = new Fraction($b->numerator, $b->denominator);

    return $fractions;
}

public function reduceOn(Fraction $a)
{
    $gcd            = $this->greatestCommonDivisorOn($a->numerator, $a->denominator);
    $a->numerator   = $a->numerator / $gcd;
    $a->denominator = $a->denominator / $gcd;

    return $a;
}
