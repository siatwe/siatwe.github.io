  • public
    • Public methods and attributes can be called everywhere in the code
  • protected
    • Access only within the class or extending classes
  • private
    • Access only inside the class
class User
{
    public $name;
    protected $email;
    private $password;

    public function setEmail($newEmail)
    {
        //Check if $email is a valid email address
        if ($this->isValidEmail($newEmail)) {
            //if yes, set attribute $email
            $this->email = $newEmail;
        }
    }

    public function getEmail()
    {
        //Returns the email address
        return $this->email;
    }

    public function setRandomPassword()
    {
        //Creates a random number between 1 and 10
        $randomPassword = strval(rand(1, 10));
        //Saves the numbers as $password, after hashPassword encodes the number to sha1
        $this->password = $this->hashPassword($randomPassword);

        return $randomPassword;
    }

    public function isCorrectPassword($password)
    {
        //Checks if $password is equal to given password
        return $this->hashPassword($password) == $this->password;
    }

    protected function isValidEmail($email)
    {
        //Checks if the given email address is valid, if yes, return it
        return filter_var($email, FILTER_VALIDATE_EMAIL) !== false;
    }

    private function hashPassword($password)
    {
        //Encodes given variable und returns it
        return sha1($password);
    }
}

$simon = new User();
$simon->name = "Simon Weis";
$simon->setEmail("simon@web.de");
