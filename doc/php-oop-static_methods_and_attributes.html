Statische Methoden und Eigenschaften, sind nicht an ein einzelnes Objekt gekoppelt, sondern sie gelten global. Um eine statische Methode oder ein Attribut aufzurufen, muss die Klasse nicht instanziert werden.

Example

class User
{
    public $name;

    //Erstellt ein statisches Array
    public static $allUsers = [];

    //Gibt die Anzahl der User die in diesem Array gespeichert wurden zurueck
    public static function numberOfUsers()
    {
        return count(self::$allUsers);
    }

    public static function createNewUser($name)
    {
        if (strlen($name) < 10) {
            echo "Nur User mit langen Namen sind erlaubt!
";
            return null;
        } else {
            $user = new User();
            $user->name = $name;

            self::$allUsers[] = $user;

            return $user;
        }
    }
}

//Erstellt eines neues Objekt der Klassse User mit der statischen Methode createNewUser
User::createNewUser("Max Mustermann");
//Gibt die Anzahl der Objekte der Klasse User ueber die statische Methode numberOfUsers aus
echo "Anzahl Nutzer: " . User::numberOfUsers() . "
";
User::createNewUser("Simon Attila Weis");
echo "Anzahl Nutzer: " . User::numberOfUsers() . "
";
User::createNewUser("Lisa Kurz");
echo "Anzahl Nutzer: " . User::numberOfUsers() . "
";
