doctrine fetching objects

created: 20170801

back

After you have persist an object to your database with doctrine, its easy to fetch it back out.

Example Controller:

/**
* @Route("/show/{articleId}", name="show")
 */
public function showAction($articleId)
{
    $article = $this->getDoctrine()
        ->getRepository(Article::class)
        ->find($articleId);

    if (!$article) {
        throw $this->createNotFoundException(
            'No article found for id '.$articleId
        );
    }

    return $this->render('default/index.html.twig', ['article' => $article]);
}

Example Twig Template:

{% extends 'base.html.twig' %}

{% block body %}
    {{ article.headline }}
    {{ article.code | raw }}
{% endblock %}
