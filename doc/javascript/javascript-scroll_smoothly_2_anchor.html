$(document).ready(function () {
    // class- / id-name
    $(".index-arrow").click(function () {
        $('html, body').animate({
            // target class- / id-name
            scrollTop: $(".index").offset().top
        }, 600);
    });
});

$('.scroll').on('click', function (e) {
    e.preventDefault();
    var href = $(this).attr('href');
    $('html, body').stop().animate({
    scrollTop: $(href).offset().top
    }, 600);
});
