Eine Route ist im Grunde eine Karte/Verweis von einem URL-Pfad zu einem Controller. Die Angabe der gewuenschten Route erfolgt ueber der Action Methode und wird Annotation genannt. Die Annotation wird innerhalb eines Kommentarblocks geschrieben und mit '@' begonnen.

Example

namespace AppBundle\Controller;

use Symfony\Bundle\FrameworkBundle\Controller\Controller;
use Sensio\Bundle\FrameworkExtraBundle\Configuration\Route;

class BlogController extends Controller
{
    /**
     * @Route("/blog", name="blog_list")
     */
    public function listAction()
    {
        ...
    }

    /**
     * @Route("/blog/{slug}", name="blog_show")
     */
    public function showAction($slug)
    {
        ...
    }
}
